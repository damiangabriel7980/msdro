var lockSchedule = {};

    /**
     * Function that creates un-parallel scheduled jobs for sending newsletter
     *
     * @name createUnparallelScheduledJob
     * @function
     * @param {String} name - The name of the scheduled job
     * @param {Object} cron - An object containing a chronometer for running scheduled jobs
     * @example
     *  {
     *       atMidnight: '0 0 0 * * *',
     *       everyHour: '* /1 * * * *',
     *       everyMinute: '/1 * * * * *'
     *  };
     * @param {Function} taskFuncPointer - One of the functions from newsLetterModule to run when the chronometer reaches a point
     *
     */
    function createUnparallelScheduledJob(name, cron, taskFuncPointer) {
        return schedule.scheduleJob(cron, function () {
            if(!lockSchedule[name]){
                lockSchedule[name] = true;
                logger.warn("job started - "+name);
                var task = taskFuncPointer();
                if(task.then){
                    task.then(
                        function (success) {
                            if(success) logger.warn(success);
                            logger.warn("job finished - "+name);
                            lockSchedule[name] = false;
                        },
                        function (err) {
                            logger.warn(err);
                            lockSchedule[name] = false;
                        }
                    );
                }else{
                    lockSchedule[name] = false;
                }
            }
        });
    }
};
